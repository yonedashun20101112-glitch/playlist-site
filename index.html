function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { try { // ⚠️【スタンドアロン対応】ここにスプレッドシートのURLからコピーしたIDを貼り付けてください！ var SPREADSHEET_ID = "1xGzpQzVBjp40YdD1y5tkdy3AR6XwJNypWOUQkEgzRTw"; var params = {}; // パラメータの解読 if (e && e.parameter) { for (var key in e.parameter) { params[key] = e.parameter[key]; } } if (e && e.postData && e.postData.contents) { var rawContent = e.postData.contents; try { var jsonParams = JSON.parse(rawContent); for (var key in jsonParams) { params[key] = jsonParams[key]; } } catch (jsonErr) { var pairs = rawContent.split('&'); for (var i = 0; i < pairs.length; i++) { var pair = pairs[i].split('='); if (pair.length === 2) { var k = decodeURIComponent(pair[0].trim()); var v = decodeURIComponent(pair[1].trim()).replace(/\+/g, ' '); params[k] = v; } } } } // スプレッドシートをID接続で開く var ss = SpreadsheetApp.openById(SPREADSHEET_ID); if (!ss) { return createJsonResponse({ status: "error", message: "スプレッドシートを開く権限がないか、IDが間違っています。" }); } // 🌟【大改良】action=getData という要求が届いた場合、スプレッドシートの全データをJSONで瞬時に返す var action = params.action || ""; if (action === "getData") { var sheet2025 = ss.getSheetByName("週間楽曲2025"); var sheet2026 = ss.getSheetByName("週間楽曲2026"); var sheetSched = ss.getSheetByName("スケジュール"); var data2025 = sheet2025 ? sheet2025.getDataRange().getValues() : []; var data2026 = sheet2026 ? sheet2026.getDataRange().getValues() : []; var dataSched = sheetSched ? sheetSched.getDataRange().getValues() : []; return createJsonResponse({ status: "success", action: "getData", tracks2025: data2025, tracks2026: data2026, schedules: dataSched }); } // 通常の「新規曲の登録」処理 var year = params.year || ""; var title = params.title || ""; var artist = params.artist || ""; var img = params.img || ""; // カバー画像URL (L列) var link = params.link || ""; // 楽曲リンク (M列) var outTrack = params.outTrack || ""; // 抹消（OUT）させたい曲情報 if (!title || !artist || !year) { return createJsonResponse({ status: "error", message: "送信されたデータが不足しています。年: " + year + ", 曲名: " + title + ", アーティスト: " + artist }); } var sheetName = "週間楽曲" + year; var sheet = ss.getSheetByName(sheetName); if (!sheet) { return createJsonResponse({ status: "error", message: year + "年度のシート「" + sheetName + "」が見つかりません。スプレッドシートのタブ名を確認してください。" }); } var today = new Date(); var formattedDate = Utilities.formatDate(today, "GMT+9", "MM/dd"); var colAValues = sheet.getRange("A:A").getValues(); var colBValues = sheet.getRange("B:B").getValues(); var colHValues = sheet.getRange("H:H").getValues(); // H列: 抹消日 // 指定した曲を同時に抹消（OUT）する処理 var outRowNumber = -1; if (outTrack && outTrack.indexOf("|||") !== -1) { var parts = outTrack.split("|||"); var outTitle = parts[0].trim().toLowerCase(); var outArtist = parts[1].trim().toLowerCase(); for (var k = 3; k < colAValues.length; k++) { if (colAValues[k][0] && colBValues[k][0]) { var currentTitle = colAValues[k][0].toString().trim().toLowerCase(); var currentArtist = colBValues[k][0].toString().trim().toLowerCase(); var isCurrentActive = (!colHValues[k] || !colHValues[k][0] || colHValues[k][0].toString().trim() === "" || colHValues[k][0].toString().trim() === "0"); if (currentTitle === outTitle && currentArtist === outArtist && isCurrentActive) { var rowNum = k + 1; sheet.getRange(rowNum, 8).setValue(formattedDate); // H列に本日の抹消日をセット outRowNumber = rowNum; break; } } } } // A列（曲名）が空いている最初の行を4行目から探す var targetRow = -1; for (var i = 3; i < colAValues.length; i++) { var val = colAValues[i][0]; if (val === undefined || val.toString().trim() === "") { targetRow = i + 1; break; } } if (targetRow === -1) { targetRow = colAValues.length + 1; } // スプレッドシートの対象行にピンポイントでセルに書き込む sheet.getRange(targetRow, 1).setValue(title); // A列: 曲名 sheet.getRange(targetRow, 2).setValue(artist); // B列: アーティスト名 sheet.getRange(targetRow, 7).setValue(formattedDate); // G列: 登録日 (07/01) if (img) { sheet.getRange(targetRow, 12).setValue(img); // L列: カバー画像URL } if (link) { sheet.getRange(targetRow, 13).setValue(link); // M列: 楽曲リンク } SpreadsheetApp.flush(); return createJsonResponse({ status: "success", message: "登録に成功しました。", row: targetRow, outRow: outRowNumber, data: { year: year, title: title, artist: artist, img: img, link: link } }); } catch (error) { return createJsonResponse({ status: "error", message: "サーバーエラーが発生しました: " + error.toString() }); } } function createJsonResponse(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
